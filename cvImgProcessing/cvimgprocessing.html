Loading Image

          ''' STEP 1 : Import library

            1. Import OpenCV via import cv2 to use OpenCV.
            2. Import MatplotLib via import matplotlib.pyplot as plt to view image.
            3. Import Argparse via import argparse it will help in passing image path in command prompt.
          '''


          import cv2
          import matplotlib.pyplot as plt
          import argparse



          ''' STEP 2 : How to load image
              1. Directly in opencv
              2. Via command prompt
              choose one which suites you.
          '''

          # Load image directly

          img = cv2.imread("fruitbowl.jpg")
          assert img is not None  # Check if image is loaded with success or Not.


          # Load image via Command Prompt

          parser = argparse.ArgumentParser()
          # if image path is not given then it will use default image path
          parser.add_argument('--path', default='fruitbowl.jpg',help='Image path.')
          params = parser.parse_args()
          img = cv2.imread(params.path)
          assert img is not None  # Check if image is loaded with success or Not.

          ''' In command prompt run below command from Project Root Directory
              python3 Basic1.py --path ../images/dog.jpeg
          '''


          ''' STEP 3 : View Loaded image
              1. Directly in opencv
              2. Via command prompt
              choose one which suites you.
          '''

          plt.imshow(img)
          plt.show()

          '''
            Points to Remember

            1. When we try to load image in OpenCV and if path is incorrect it will not complain
               hence it is good idea to check if file is loaded successfully or not.

Basic Operations

'''
Created on 04-Jan-2019
@author: http://www.eduliant.com

All codes are tested with

OPencv - 4.0.0-dev
Python - 3.5.2
Ubuntu - 16.04 LTS

'''

import cv2
import matplotlib.pyplot as plt
import argparse

parser = argparse.ArgumentParser()
parser.add_argument('--path', default='../images/fruitbowl.jpg',help='Image path.')
params = parser.parse_args()
img = cv2.imread(params.path)
assert img is not None  # Check if image is loaded with success or Not.


plt.imshow(img)
plt.show()

'''
    Operation 1 : Convert Image from BGR to RGB format.
    When we load image in OpenCV it loads image in BGR, not RGB, format.
    hence we need to convert it to RGB format.

'''

img = cv2.cvtColor(img,cv2.COLOR_BGR2RGB)

plt.imshow(img)
plt.show()

'''
    Operation 2 : Display features of loaded image
    Like its shape and datatype

'''

print('read {}'.format(params.path)) # gives path of the image
print('shape:', img.shape) # gives an image's size and number of color channels
print('dtype:', img.dtype) # gives datatype of the nd array


'''
  Points to Remember

  1. The loaded image is represented as a NumPy array.
  2. OpenCV loads images in BGR, not RGB, format.
  3. shape tuple (img.shape) can be interpreted as image height, image width,color channels count.
  4. cv2.imread() also supports optional flag
      1. specify unit conversion
      2. specify image is grayscale or color

'''
