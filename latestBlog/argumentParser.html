Learning Objective : Learn how to use Argument Parser and send arguments form command line.
Assumptions :Python is already installed.

              '''

              '''Step 1: import argparse
              '''

              import argparse


              ''' Step 2: Create an object of argparse and then add argument
              '''

              ap = argparse.ArgumentParser()
              ap.add_argument("-p","--program",required=True,help="It will print program name entered by you")
              ap.add_argument("--text",required=False,help="It will print optional message")


              ''' Step 3: turn the parsed command line arguments into python dictionary
                          where key to the dictionary is the name of the command line argument
                          and value is supplied value.
              '''

              args = vars(ap.parse_args())


              ''' Step 4: Sample python code using command line arguments
              '''

              for n in args["program"]:
                  print(n)

              print ("optional message : ",args["text"])
